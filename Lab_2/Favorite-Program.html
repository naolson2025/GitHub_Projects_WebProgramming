Contact Creator and Editor

My program allows a user to create a csv file with python that contains contacts. 
import csv
            def main():
                try:
                    create()
                    input_contact()
                    output_table()
                except:
                    print("You must enter valid text.")


            def create():
                create_file = open('contacts.csv', 'w', newline='')            # Open the file contacts.csv in write mode
                title_writer = csv.writer(create_file)                         # assign that file to a variable
                title_writer.writerow(['Name', 'address', 'Email', 'Phone'])              # write the headers
                create_file.close()


            def input_contact():
                append_file = open('contacts.csv', 'a', newline='')            # open the file to append
                contact_writer = csv.writer(append_file)
                count = 1
                for contact in range(6):                                       # loop to let the user add contacts
                    print("Enter contact number %d's information" % count)
                    contact_list = [input("Name: "), input("Street Address: "), input("Email: "), input("Phone: ")]
                    contact_writer.writerow(contact_list)
                    count += 1
                append_file.close()                                            # close the file


            def output_table():
                read_file = open('contacts.csv')                               # open the file output.csv
                spreadsheet = csv.reader(read_file)                            # assign a variable and read the file
                for row in spreadsheet:                                        # loop through the rows in the csv file
                    print('{:<10} {:<20} {:<20} {:<10}'.format(*row))               # print each row with spacing


            main()
